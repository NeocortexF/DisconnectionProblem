private String getNames(int key) {
        //метод возвращает имя человека по должности по ключу 1,2,3 из базы MySQL
        String name = null;

        String conString = "jdbc:mysql://localhost:3306/databaseName?autoReconnect=true&useSSL=false";
        String username = "username";
        String password = "password";
        String sql = "";

        switch (key) {
            case 1:
                sql = "SELECT * FROM escadra2 WHERE id = 1";
                break;
            case 2:
                sql = "SELECT * FROM escadra2 WHERE id = 2";
                break;
            case 3:
                sql = "SELECT * FROM escadra2 WHERE id = 3";
                break;
        }
        try {
            Connection con = DriverManager.getConnection(conString, username, password);
            Statement s = con.prepareStatement(sql);
            ResultSet rs = s.executeQuery(sql);
            while (rs.next()) {
                name = rs.getString(2);
                return name;
            }
            if (rs != null) rs.close();
            if (s != null) s.close();
            if (con != null) con.close();
        } catch (Exception ex) {
            ex.printStackTrace();
        } 
        return name;
    }
